## 🚀 使用方法

### 基本使用
```javascript
const { XMRig } = require('node-xmrig');

const miner = new XMRig({
    cpu: {
        threads: 4  // 指定使用 4 個 CPU 核心
    },
    pool: {
        url: 'gulf.moneroocean.stream:10128',  // 指定 pool
        user: '你的錢包地址',                   // 指定錢包地址
        pass: 'x'
    }
});

miner.on('hash', (data) => {
    console.log(`算力: ${data.rate} ${data.unit}`);
});

await miner.start();
```

### 運行測試
```bash
npm test                    # 運行測試套件
npm run example:basic      # 運行基本示例
npm run example:config     # 運行配置示例
npm run example:events     # 運行事件示例
```
